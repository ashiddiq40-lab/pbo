// dataset JSON lokal const users = [ { username: "admin", password: "12345", nama: "Administrator", role: "Admin" }, { username: "user", password: "abcde", nama: "User Biasa", role: "User" } ]; // class OOP class Login { constructor(users) { this.users = users; } cekLogin(username, password) { return this.users.find( user => user.username === username && user.password === password ); } } // object login const loginApp = new Login(users); // fungsi login function login() { const username = document.getElementById("username").value; const password = document.getElementById("password").value; const user = loginApp.cekLogin(username, password); if (!user) { alert("salah login"); } else { document.getElementById("info").innerHTML = `

Login Berhasil

Nama : ${user.nama}

Username : ${user.username}

Role : ${user.role}

`; } }
